//Creating a DATABASE

CREATE DATABASE eb;
USE eb;



//Creating a admin+customer user table

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    email VARCHAR(100),
    password VARCHAR(255),
    role VARCHAR(20) DEFAULT 'customer'
);

//Creating a bills table

CREATE TABLE bills (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    consumer_no VARCHAR(50),
    month VARCHAR(50),
    units INT,
    amount INT,
    status VARCHAR(20) DEFAULT 'Unpaid'
);



\\\\for contact

CREATE TABLE messages (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    subject VARCHAR(255),
    message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
